CREATE TABLE ARTISTE(CODE_ARTISTE INTEGER NOT NULL PRIMARY KEY,NOM VARCHAR(80) NOT NULL);


CREATE TABLE ALBUM(CODE_ALBUM INTEGER NOT NULL PRIMARY KEY,NOM VARCHAR(80) NOT NULL,ARTISTE_ID INTEGER,
CONSTRAINT FK_ALBUM FOREIGN KEY(ARTISTE_ID) REFERENCES ARTISTE(CODE_ARTISTE));


CREATE TABLE CHANSON(ID INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 1) NOT NULL PRIMARY KEY, NUMERO_CHANSON INTEGER NOT NULL, TITRE VARCHAR(80) NOT NULL, DUREE INTEGER NOT NULL,ALBUM_ID INTEGER NOT NULL,CONSTRAINT FK_CHANSON FOREIGN KEY(ALBUM_ID) REFERENCES ALBUM(CODE_ALBUM));
